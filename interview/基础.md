#### 任意数求和

```
const sum = (...args) => {
  var data = args.reduce((pre, next) => {
    next = !isNaN(next) ? Number(next) : 0;
    return pre + next
  }, 0)
  return data
}
console.log(sum(1, 2, 2, '33'))
```
